import {
  BadRequestException,
  ConflictException,
  HttpException,
  Injectable,
  InternalServerErrorException,
} from '@nestjs/common';
import { Prisma, quest, task, user } from '@prisma/db';
import { DatabaseService } from '@providers/database/database.service';
import { RedisService } from '@providers/redis/redis.service';
import { TaskContent } from 'src/types';
import { taskApiCall } from './utils';

@Injectable()
export class QuestService {
  constructor(
    private readonly databaseService: DatabaseService,
    private readonly redisService: RedisService,
  ) {}

  public async getQuests(page: number, size: number, filter: object) {
    const whereCluster: Prisma.questFindManyArgs['where'] = {};

    if (filter) {
      if (filter['title']) {
        whereCluster.title = {
          contains: filter['title'],
        };
      }
    }

    try {
      const [count, list] = await this.databaseService.$transaction([
        this.databaseService.quest.count({ where: whereCluster }),
        this.databaseService.quest.findMany({
          where: whereCluster,
          skip: (page - 1) * size,
          take: size,
        }),
      ]);

      const response = {
        count,
        list,
      };

      return response;
    } catch (err) {
      throw new InternalServerErrorException('error.INTERNAL_SERVER_ERROR');
    }
  }

  public async getQuest(questId: string) {
    const cachedQuest = await this.redisService.get(`quest:${questId}`);

    if (cachedQuest) {
      return JSON.parse(cachedQuest);
    }

    try {
      const quest = await this.databaseService.quest.findUnique({
        where: {
          id: questId,
        },
        include: {
          tasks: true,
        },
      });

      if (!quest) {
        throw new BadRequestException('error.QUEST_NOT_FOUND');
      }

      await this.redisService.set(
        `quest:${questId}`,
        JSON.stringify(quest),
        86400,
      );

      return quest;
    } catch (err) {
      if (err instanceof HttpException) {
        throw err;
      }
      throw new InternalServerErrorException('error.INTERNAL_SERVER_ERROR');
    }
  }

  public async getQuestUserStatus(questId: string, userId: string) {
    try {
      const userQuestStatus =
        await this.databaseService.userQuestStatus.findUnique({
          where: {
            unique_user_quest: {
              userId,
              questId,
            },
          },
        });

      return userQuestStatus
        ? { status: userQuestStatus.status }
        : { status: false };
    } catch (err) {
      throw new InternalServerErrorException('error.INTERNAL_SERVER_ERROR');
    }
  }

  public async finishQuestTask(
    userId: string,
    questId: string,
    taskId: string,
  ) {
    let user: user;
    let quest: quest & { tasks: task[] };
    let task: task;

    // Fetch user, quest and task
    try {
      user = await this.databaseService.user.findUnique({
        where: {
          id: userId,
        },
      });

      if (!user) {
        throw new BadRequestException('error.USER_NOT_FOUND');
      }

      quest = await this.databaseService.quest.findUnique({
        where: {
          id: questId,
        },
        include: {
          tasks: true,
        },
      });

      if (!quest) {
        throw new BadRequestException('error.QUEST_NOT_FOUND');
      }

      task = quest.tasks.find((task) => task.id === taskId);

      if (!task) {
        throw new BadRequestException('error.TASK_NOT_FOUND');
      }
    } catch (err) {
      if (err instanceof HttpException) {
        throw err;
      }
      throw new InternalServerErrorException('error.INTERNAL_SERVER_ERROR');
    }

    // Check user already finished the task
    const userTaskStatus = await this.databaseService.userTaskStatus.findUnique(
      {
        where: {
          unique_user_task: {
            userId,
            taskId,
          },
        },
      },
    );

    if (userTaskStatus) {
      throw new ConflictException('error.TASK_ALREADY_FINISHED');
    }

    // Handle task finish
    const response = await this.handleTaskFinish(task, user.walletAddress);
    if (!response) {
      throw new BadRequestException('error.TASK_NOT_FINISHED');
    }

    // Finish the task
    try {
      const [_, userQuestStatus, __, history] =
        await this.databaseService.$transaction([
          this.databaseService.userTaskStatus.create({
            data: {
              userId,
              taskId,
              status: true,
            },
          }),
          this.databaseService.userQuestStatus.upsert({
            where: {
              unique_user_quest: {
                userId,
                questId,
              },
            },
            update: {
              clearedTaskCount: {
                increment: 1,
              },
            },
            create: {
              userId,
              questId,
              clearedTaskCount: 1,
            },
          }),
          this.databaseService.leaderboard.update({
            where: {
              userId: userId,
            },
            data: {
              point: {
                increment: task.rewardPoint,
              },
            },
          }),
          this.databaseService.userLeaderboardHistory.create({
            data: {
              userId,
              taskId,
              title: task.title + ' task Completed',
              point: task.rewardPoint,
            },
          }),
        ]);

      // Check if all required tasks are finished
      const requiredTaskCount = quest.tasks.filter(
        (task) => task.type === 'REQUIRED',
      ).length;

      await this.databaseService.userQuestStatus.update({
        where: {
          unique_user_quest: {
            userId,
            questId,
          },
        },
        data: {
          status: userQuestStatus.clearedTaskCount === requiredTaskCount,
        },
      });

      return history;
    } catch (err) {
      if (err instanceof Prisma.PrismaClientKnownRequestError) {
        if (err.code === 'P2002') {
          throw new ConflictException('error.TASK_ALREADY_FINISHED');
        }
      }

      if (err instanceof HttpException) {
        throw err;
      }

      throw new InternalServerErrorException('error.INTERNAL_SERVER_ERROR');
    }
  }

  // Utility function
  private async handleTaskFinish(
    task: task,
    userWalletAddress: string,
  ): Promise<boolean> {
    const contentType = task.contentType;
    const content = task.content as TaskContent;
    let response: boolean = false;

    switch (contentType) {
      case 'SNS': {
        response = true;
        break;
      }
      case 'API': {
        response = await taskApiCall(content, userWalletAddress);
        break;
      }
    }

    return response;
  }
}
